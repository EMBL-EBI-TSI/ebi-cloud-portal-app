var credentials_1 = require('../services/credentials/credentials');
var Login = (function () {
    function Login(router, account, credentials) {
        this.router = router;
        this.account = account;
        this.credentials = credentials;
    }
    Login.prototype.login = function (event, username, password) {
        var _this = this;
        event.preventDefault();
        this.credentials.setCredentials(username, password);
        this.account.getAccount(this.credentials)
            .subscribe(function (data) {
            console.log('User data is %O', data);
            _this.router.parent.navigateByUrl('/home');
        }, function (err) {
            console.log(err);
            _this.credentials.clearCredentials();
            _this.router.parent.navigateByUrl('/login');
        }, function () {
            console.log('Authentication Complete');
        });
    };
    Login.prototype.ngOnInit = function () {
        console.log('hello *Login* component');
    };

        core_1.Component({

            providers: [

                credentials_1.Credentials

            template: require('./login.html')
        }), 
        __metadata('design:paramtypes', [router_1.Router, account_1.Account, credentials_1.Credentials])
    ], Login);
    return Login;
})();
